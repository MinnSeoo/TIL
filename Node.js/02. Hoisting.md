## **Hoisting**

---

```jsx

console.log(Hello);     
// 아직 생성하지 않은 Hello 변수를 출력하려고 함 -> 결과값 : undefind
// 위 코드가 오류를 발생시키지 않는 이유 -> 호이스팅 때문

var Hello = 'Hello';   
 // JavaScript에서는 변수 생성의 선언 단계와 할당 단계로 나뉘게 된다.
// 선언 부분은 변수를 생성하고 나서 호이스팅을 통해 현재 범위의 맨 위로 올라가게 된다. (그리고 초기값이 할당된다.)

// 위 계념을 이 코드에 대입시키면 출력결과로 undefind (초기값)이 출력된다.

// 함수 호이스팅

func();                 
// 함수 호이스팅는 var 호이스팅과 비슷한 원리로 동작한다.

// 함수를 선언하기 전 먼저 호출하면 메모리에 함수 선언부분이 로드되고 그 다음에 호출되기 때문에 ( = 호이스팅) 

// 결과값으로 hosting test가 출력이 된다. 
function func() {       
    console.log("hoisting test");
}
```
